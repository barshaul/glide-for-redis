def publish_messages():
    global channels, publisher_error_count
    while True:
        try:
            timestamp = datetime.now(timezone.utc).strftime("%H:%M:%S")
            for idx, channel in enumerate(channels):
                message = f"channel_{idx+1}_{timestamp}"
            publisher.publish(channel, message)
            publisher_messages.appendleft(message)
        except Exception as e:
        print(f"Publisher received error {e}")
            publisher_error_count += 1
        time.sleep(1)

def create_new_subscriber() -> PubSubClient:
    global channels, subscriber_error_count, host, port
    while True:
        try:
            new_subscriber = other_client.create(host, port).pubsub()
            new_subscriber.subscribe(channels)
            return new_subscriber
        except Exception as e:
            print(f"Failed to create subsriber {e}")
            subscriber_error_count += 1
            time.sleep(0.5)


def listen_on_channels():
    global subscriber, subscriber_error_count
    while True:
        try:
            msg = subscriber.get_message()
            if msg:
                message = msg["data"]
                subscriber.appendleft(message)
        except Exception as e:
            print(f"Subscriber received error {e}")
            subscriber_error_count += 1
            subscriber = create_new_subscriber()
